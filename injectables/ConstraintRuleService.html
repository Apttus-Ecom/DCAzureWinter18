Returns : Observable<string>

Public getConstraintRules
getConstraintRules(product: Product)

getConstraintRules is the default service to get a list of constraint rules for a given product. Currently only matches constraint rules for the product itself or the product group it is a member of.

Example:

      import { ConstraintRuleService, Product } from '@apttus/ecommerce';
      import { Observable } from 'rxjs/Observable';

      export class MyComponent implements OnInit{
           constraintRules: Array<ConstraintRule>;
           constraintRules$: Observable<Array<ConstraintRule>>;

           constructor(private constraintRuleService: ConstraintRuleService){}

           getConstraintRulesForProduct(product: Product){
                this.constraintRuleService.getConstraintRules(product).subscribe(p => this.constraintRules = p);
                // or
                constraintRules$ = this.constraintRuleService.getConstraintRules(product);
           }
      }
