Public getProductOptions
getProductOptions(productCode: string, groupName: string, pricing: boolean)

Method gets a list of product option components for a given product

Example:

      import { ProductOptionService, ProductOptionComponent } from '@apttus/ecommerce';
      import { Observable } from 'rxjs/Observable';

      export class MyComponent implements OnInit{
           optionComponentList: Array<ProductOptionComponent>;
           optionComponentList$: Observable<Array<ProductOptionComponent>>;

           constructor(private productOptionService: ProductOptionService){}

           getProductOptions(productCode: string){
                this.productOptionService.getProductOptions(productCode).subscribe(o => this.optionComponentList = o);
                // or
                this.optionComponentList$ = this.productOptionService.getProductOptions(productCode);
           }
      }
Parameters :
