import { Injectable } from '@angular/core';
import { SObjectService, ChildRecord, SObjectType, SoapService } from 'ng-salesforce';
import { Cart, UserBase, User, ProductAttributeValue, Product, CartItem } from '../classes/index';
import { UserService, PriceListService, AccountService, ProductAttributeValueService } from './index';
import { ProductOptionForm } from '../interfaces/product-option-form.interface';
import { Observable } from 'rxjs/Rx';
import * as _ from 'lodash';

@Injectable({
    providedIn: 'root'
})
@SObjectType(CartItem)
export class CartItemService extends SObjectService{}

@Injectable({
    providedIn: 'root'
})
@SObjectType(Cart)
export class CartService extends SObjectService {

    private userService = this.injector.get(UserService);
    private plService = this.injector.get(PriceListService);
    private accountService = this.injector.get(AccountService);
    private cartItemService = this.injector.get(CartItemService);
    private attrValueService = this.injector.get(ProductAttributeValueService);
    private soapService = this.injector.get(SoapService);

    getMyCart(): Observable<Cart> {
        return this.cacheService.get('cart', () => this.loadCart());
    }

    private loadCart(): Observable<Cart> {
        let cartData = null;
        return Observable.combineLatest(this.userService.me(), this.accountService.getCurrentAccount(), this.plService.getEffectivePriceListId(), this.plService.getPriceListId())
            .distinctUntilChanged()
            .filter(res => (
                (res[1].Price_List__c  === null || res[1].Price_List__c  === res[2])
                && ((_.get(res, '[0].Contact.Id')) ?
                    (res[0].Contact.AccountId  === res[1].Id || res[0].Contact.AccountId  === res[1].ParentId)
                    : true))
            )
            .flatMap(([user, account, epli, pli]) => {
                if(!epli || !pli)
                    return Observable.throw(new Error('A price list could not be found for your cart.'));

                let smallUser = new UserBase();
                smallUser.Alias = user.Alias;
                cartData = Object.assign(this.getInstance(), new Cart(pli, epli, account.Id, user));
                let key = user.Id + ':' + account.Id + ':' + pli + ':' + epli;
                if (user.Id && account.Id && pli && epli) {
                    return this.syncLocalCartWithUser().flatMap(() => this.queryBuilder(
                        `OwnerId = '` + user.Id + `' AND Apttus_Config2__AccountId__c = '` + account.Id + `' AND Apttus_Config2__PriceListId__c = '` + pli + `' AND Apttus_Config2__EffectivePriceListId__c = '` + epli + `' AND Apttus_Config2__Status__c = 'New'`
                        , 1
                        , 0
                        , 'CreatedDate'
                        , 'DESC'
                        , true));
                } else if (pli && epli && this.cacheService._get('local-cart')) {
                    return this.queryBuilder(
                        `Id = '` + this.cacheService._get('local-cart') + `' AND Apttus_Config2__PriceListId__c = '` + pli + `' AND Apttus_Config2__EffectivePriceListId__c = '` + epli + `' AND Apttus_Config2__Status__c = 'New'`
                        , 1
                        , 0
                        , 'CreatedDate'
                        , 'DESC'
                        , true);
                }

                // Return null if neither
                else
                    return Observable.of(null);

            }).flatMap((cartList: Array<Cart>) => {
                if ((cartList  === null || cartList.length  === 0) && cartData) {
                    return this.create([cartData])
                        .flatMap((res) => {
                            console.log(res);
                            if (cartData.Owner.Alias === 'guest')
                                this.cacheService._set('local-cart', res[0], true);

                            return this.get([res[0]]);
                        }).map(res => res[0]);
                } else if (cartList)
                    return Observable.of(cartList[0]);
                else
                    return Observable.of(cartData);
            });
    }

    public syncLocalCartWithUser(): Observable<void> {
        if (this.cacheService._get('local-cart')) {
            return Observable.combineLatest(
                this.userService.me().take(1),
                this.plService.getPriceListId().take(1),
                this.plService.getEffectivePriceListId().take(1),
                this.accountService.getCurrentAccount().take(1))
                .flatMap(([user, plId, eplId, account]) => {
                    if (user.Id && plId && eplId && account.Id)
                        return this.forceService.post('setCartOwner', {
                            cartId: this.cacheService._get('local-cart'),
                            userId: user.Id,
                            priceListId: plId,
                            accountId: account.Id,
                            effectivePriceListId: eplId
                        })
                            .catch(e => {
                                return Observable.of(null);
                            })
                            .map(() => this.cacheService.clearKey('local-cart'));
                    else
                        return Observable.of(null);
                });
        } else {
            return Observable.of(null);
        }
    }


    /**
     * addProductToCart:   Method takes a product and a quantity and updates the cart
     * @param product      Product object for method
     * @param quantity     number value for quantity
     * @return Observable  Returns an empty observable when complete
     */
    public addProductToCart(product: Product,
        quantity: number = 1,
        merge: boolean = true,
        attributeValues?: Array<ProductAttributeValue>,
        productOptions?: Array<ProductOptionForm>,
        quick: boolean = false,
        rollBackOnFail: boolean = true,
        timeout: number = 10000): Observable<void> {
        if (quantity <= 0)
            return Observable.throw('Cannot add a quantity less than 1');

        return this.getMyCart()
            .filter(cart => cart._state !== 'calculating')
            .take(1).flatMap(cart =>
                this.plService.getPriceListId()
                    .flatMap(priceListId => {
                        let cartItem: CartItem = Object.assign({}, this.cartItemService.getInstance() as CartItem);

                        let idx;
                        let temp;
                        if (_.get(cart, 'Apttus_Config2__LineItems__r.records')) {
                            idx = _.findIndex(cart.Apttus_Config2__LineItems__r.records, (r) => r.Apttus_Config2__ProductId__c === product.Id);
                            temp = cart.Apttus_Config2__LineItems__r.records[idx];
                        }

                        cartItem.Apttus_Config2__ProductId__r = product;
                        cartItem.Apttus_Config2__ProductId__c = product.Id;
                        cartItem.Apttus_Config2__PriceListId__c = priceListId;
                        cartItem.Apttus_Config2__ListPrice__c = -1;

                        if (temp && merge && product.Apttus_Config2__ConfigurationType__c !== 'Bundle') {
                            cartItem.Id = temp.Id;
                            cartItem.Apttus_Config2__Quantity__c = Number(temp.Apttus_Config2__Quantity__c) + Number(quantity);
                            if (_.get(cart, 'Apttus_Config2__LineItems__r.records'))
                                cart.Apttus_Config2__LineItems__r.records.splice(idx, 1, cartItem);
                            else
                                cart.Apttus_Config2__LineItems__r = {
                                    records: [cartItem]
                                } as ChildRecord;
                            return this.cartItemService.update([cartItem]);
                        } else {
                            const primaryLineNumber = cart.Apttus_Config2__NumberOfItems__c + 1;
                            const newItems: Array<CartItem> = [Object.assign(cartItem, {
                                Apttus_Config2__ItemSequence__c: primaryLineNumber,
                                Apttus_Config2__LineNumber__c: primaryLineNumber,
                                Apttus_Config2__PrimaryLineNumber__c: primaryLineNumber,
                                Apttus_Config2__ConfigurationId__c: cart.Id,
                                Apttus_Config2__Quantity__c: quantity,
                                Apttus_Config2__LineType__c: 'Product/Service'
                            }) as CartItem];


                            // Create the options
                            let i = 1;
                            if(productOptions){
                                productOptions.forEach(option => {
                                    newItems.push({
                                        Apttus_Config2__ConfigurationId__c: cart.Id,
                                        Apttus_Config2__ItemSequence__c: primaryLineNumber + i,
                                        Apttus_Config2__LineNumber__c: primaryLineNumber + i,
                                        Apttus_Config2__PrimaryLineNumber__c: primaryLineNumber + i,
                                        Apttus_Config2__Quantity__c: option.quantity,
                                        Apttus_Config2__LineType__c: 'Option',
                                        Apttus_Config2__ProductId__r: product,
                                        Apttus_Config2__ProductId__c: product.Id,
                                        Apttus_Config2__PriceListId__c: priceListId,
                                        Apttus_Config2__ListPrice__c: -1,
                                        Apttus_Config2__ProductOptionId__c: option.productOptionComponent.Id,
                                        Apttus_Config2__OptionId__c: option.productOptionComponent.Apttus_Config2__ComponentProductId__c
                                    } as CartItem);
                                    i++;
                                });
                            }

                            if (_.get(cart, 'Apttus_Config2__LineItems__r.records'))
                                cart.Apttus_Config2__LineItems__r.records.concat(newItems);
                            else
                                cart.Apttus_Config2__LineItems__r = {
                                    records: newItems
                                } as ChildRecord;
                            return this.cartItemService.create(newItems);
                        }
                    })
                    .flatMap((res: Array<string>) => {
                        if (attributeValues && attributeValues.length > 0) {
                            attributeValues.forEach(attr => attr.Apttus_Config2__LineItemId__c = res[0]);
                            return this.attrValueService.create(attributeValues).map(attrs => res[0]);
                        } else
                            return Observable.of(res[0]);
                    })
                    .flatMap(cartItemId => {
                        if (quick) {
                            this.priceCart(cartItemId, timeout, rollBackOnFail).subscribe(() => { });
                            return Observable.of(null);
                        } else
                            return this.priceCart(cartItemId, timeout, rollBackOnFail);
                    })
            );
    }

    public updateQuantity(cartItemList: Array<CartItem>): Observable<void> {
        let _temp = new Array<CartItem>();
        cartItemList.forEach(item => {
            let newItem = Object.assign({}, this.cartItemService.getInstance() as CartItem);
            newItem.Apttus_Config2__Quantity__c = item.Apttus_Config2__Quantity__c;
            newItem.Apttus_Config2__PricingStatus__c = 'Pending';
            newItem.Id = item.Id;
            _temp.push(newItem);
        });
        return this.cartItemService.update(_temp).flatMap(() => this.priceCart());
    }

    public removeCartItem(_cartItem: CartItem, quick: boolean = false): Observable<void> {
        return this.getMyCart().take(1).flatMap(cart => {
            return this.cartItemService.delete([_cartItem])
                .flatMap(() => {
                    if (quick) {
                        let idx = _.findIndex(cart.Apttus_Config2__LineItems__r.records, (r) => r.Id  === _cartItem.Id);
                        cart.Apttus_Config2__LineItems__r.records.splice(idx, 1);
                        this.priceCart().subscribe(() => { });
                        return Observable.of(null);
                    } else
                        return this.priceCart();
                });
        });
    }

    public priceCart(cartItemId: string = null, timeout: number = 10000, rollBackOnFail: boolean = true): Observable<void> {
        return this.getMyCart().take(1)
            .flatMap(cart => {
                cart._state = 'calculating';
                this.cacheService.republish('cart', Object.assign({}, cart));

                return this.forceService.post('repriceCart', {cartId : cart.Id})
                    .timeout(timeout)
                    .catch(e => {
                        if (cartItemId && rollBackOnFail) {
                            const c = Object.assign({}, this.cartItemService.getInstance());
                            c.Id = cartItemId;
                            return this.cartItemService.delete([c]);
                        } else
                            return Observable.of(null);
                    })
                    .flatMap(res => this.cacheService.update('cart'));
            });
    }

}
