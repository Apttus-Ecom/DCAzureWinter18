import { Injectable } from '@angular/core';

import { Observable, BehaviorSubject } from 'rxjs/Rx';
import { SObjectService, SObjectType } from 'ng-salesforce';
import { PriceListService, UserService } from './index';
import { Category, Classification, Product, ProductCategory, CategoryTranslation } from '../classes/index';
import * as _ from 'lodash';


@Injectable({
    providedIn: 'root'
})
@SObjectType(Category)
export class CategoryService extends SObjectService {

    private _categories: BehaviorSubject<Array<Category>>;

    private plService = this.injector.get(PriceListService);
    private userService = this.injector.get(UserService);

    public getCategories(): Observable<Array<Category>> {

        return this.cacheService.get('categories', () =>
            this.plService.getPriceListId()
                .filter(res => (res != null))
                .flatMap(priceListId => this.userService.me().take(1)
                    .flatMap(user => {
                        return this.where(`Apttus_Config2__HierarchyId__c IN (SELECT Apttus_Config2__HierarchyId__c FROM Apttus_Config2__PriceListCategory__c WHERE Apttus_Config2__PriceListId__c = '` + priceListId + `')
                           AND (Apttus_Config2__AncestorId__c <> NULL OR Apttus_Config2__PrimordialId__c = NULL)
                           AND Apttus_Config2__IsHidden__c = FALSE`)
                            .map(categories => this.translateCategories(user.LanguageLocaleKey, categories));
                    })
                )
        );
    }

    public getCategoryTree(categoryNames?: Array<string>): Observable<Array<Category>> {
        return this.getCategories().map(data => {
            let roots = [];
            let recursive = (a) => {
                a._children = new Array<any>();

                for (let i = 0; i < data.length; i++) {
                    if (data[i].Apttus_Config2__AncestorId__c === a.Id && data[i].Apttus_Config2__AncestorId__c) {
                        let b = data[i];
                        if (!b._children)
                            recursive(b);

                        a._children.push(b);
                    }
                }
            };
            for (let record of data) {
                recursive(record);
                if (categoryNames && categoryNames.indexOf(record.Name) >= 0 && record._children && record._children.length > 0) {
                    roots = roots.concat(record);
                } else if (!categoryNames && record._children && record._children.length > 0 && !record.Apttus_Config2__AncestorId__c)
                    roots = roots.concat(record);
            }
            return roots;
        });
    }

    getCategoryBranch(categoryId: string): Observable<Array<Category>> {
        return this.getCategories().map(data => {
            let categoryArray = new Array<Category>();
            let recursive = (_categoryId) => {
                for (let category of data) {
                    if (category.Id === _categoryId) {
                        categoryArray.unshift(category);
                        if (category.Apttus_Config2__AncestorId__c)
                            recursive(category.Apttus_Config2__AncestorId__c);
                    }
                }
            };
            recursive(categoryId);
            return categoryArray;
        });
    }

    getCategoryBranchForProduct(product: Product): Observable<Array<Category>> {
        if (_.get(product, 'Apttus_Config2__Categories__r.records[0]')) {
            const validCategories = product.Apttus_Config2__Categories__r.records.filter(r => r.Apttus_Config2__ClassificationId__r.Apttus_Config2__PrimordialId__c != null || r.Apttus_Config2__ClassificationId__r.Apttus_Config2__AncestorId__c != null);
            if (validCategories && validCategories.length > 0) {
                let category: ProductCategory = validCategories[0];
                return this.getCategoryBranch(category.Apttus_Config2__ClassificationId__c);
            } else
                return Observable.of(null);

        } else
            return Observable.of(null);
    }

    getSubcategories(categoryId: string): Observable<Array<Category>> {
        if (categoryId) {
            return this.getCategories().map(data => {
                let categoryArray = new Array<Category>();
                for (let category of data) {
                    if (category.Apttus_Config2__AncestorId__c && (category.Apttus_Config2__AncestorId__c.substring(0, 15) === categoryId.substring(0, 15)))
                        categoryArray.push(category);
                }
                return categoryArray;
            });
        } else
            return Observable.of(null);
    }

    getRelatedCategories(categoryId: string): Observable<Array<Category>> {
        return this.getCategories().map(data => {
            let categoryArray = new Array<Category>();

            let ancestorId = null;
            for (let category of data) {
                if (category.Id === categoryId)
                    ancestorId = category.Apttus_Config2__AncestorId__c;
            }

            for (let category of data) {
                if (category.Apttus_Config2__AncestorId__c === ancestorId)
                    categoryArray.push(category);
            }
            categoryArray = categoryArray.sort((a, b) => {
                if (a.Name > b.Name) return 1;
                else if (a.Name < b.Name) return -1;
                else return 0;
            });

            return categoryArray.slice(0, 100);
        });
    }


    private translateCategories(language: string, _categoryList: Array<Category>): Array<Category> {
        let categoryList = _categoryList.map(x => Object.assign({}, x));
        for (let category of categoryList) {
            if (category && category.Apttus_Config2__Translation__r && category.Apttus_Config2__Translation__r.records) {
                let translation: CategoryTranslation = category.Apttus_Config2__Translation__r.records.filter((record: CategoryTranslation) => {
                    if (record && record.Apttus_Config2__Language__c)
                        return record.Apttus_Config2__Language__c.toLowerCase() === language.toLowerCase();
                    else return false;
                })[0];

                if (translation)
                    category.Apttus_Config2__Label__c = translation.Apttus_Config2__Label__c;

            }
        }
        return categoryList;
    }
}
