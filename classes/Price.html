import { Product, ProductAttributeValue, ProductOptionComponent, PriceListItem, PriceMatrix, PriceMatrixEntry, ProductAttribute } from '../classes/index';
import { PriceListService, PriceMatrixService } from './index';
import { ProductOptionForm } from '../interfaces/index';

import { Observable } from 'rxjs/Observable';
import { Injector, Inject, forwardRef } from '@angular/core';
import * as _ from 'lodash';

export class PriceService {

    constructor(@Inject(forwardRef(() => PriceListService)) private priceListService: PriceListService,
        @Inject(forwardRef(() => PriceMatrixService)) private priceMatrixService: PriceMatrixService) { }


    public getProductPrice(product: Product, quantity: number = 1, attributes?: Array<ProductAttributeValue>, options?: Array<ProductOptionForm>): Observable<Price> {
        return this.priceListService.getPriceList().flatMap(priceList => {
            const p: Price = new Price();

            const productPriceLists = _.get(product, 'Apttus_Config2__PriceLists__r.records');
            const priceListItem = productPriceLists.filter(pli => pli.Apttus_Config2__PriceListId__c === priceList.Id)[0];

            if (!priceListItem)
                return Observable.throw(new Error('Product does not have a valid price list item'));
            else {
                p.productPrice = priceListItem.Apttus_Config2__ListPrice__c;
                return this.priceMatrixService.getPriceMatrixData([priceListItem]).flatMap(priceMatrixData => {
                    p.basePrice = priceListItem.Apttus_Config2__ListPrice__c * quantity;
                    p.discountPrice = priceListItem.Apttus_Config2__ContractPrice__c * quantity;

                    // Calculate the matrices
                    attributes.forEach(attribute => this.matrixAdjustment(p, priceMatrixData, attribute, quantity));

                    // Calculate options
                    this.optionAdjustment(p, options);

                    // Check if there is a price list level discount
                    if (priceList.Apttus_Config2__BasedOnAdjustmentType__c && priceList.Apttus_Config2__BasedOnAdjustmentAmount__c)
                        this.priceListAdjustment(p, priceList.Apttus_Config2__BasedOnAdjustmentType__c, priceList.Apttus_Config2__BasedOnAdjustmentAmount__c, priceListItem);
                    return Observable.of(p);
                });
            }
        });
    }

    public getAttributePriceForProduct(product: Product, attribute: ProductAttribute, attributeValue: any): Observable<number> {
        return this.priceListService.getPriceList().flatMap(priceList => {

            const productPriceLists = _.get(product, 'Apttus_Config2__PriceLists__r.records');
            const priceListItem = productPriceLists.filter(pli => pli.Apttus_Config2__PriceListId__c === priceList.Id)[0];

            if (!priceListItem)
                return Observable.throw(new Error('Product does not have a valid price list item'));
            else {
                return this.priceMatrixService.getPriceMatrixData([priceListItem]).map(priceMatrixData => {
                    if (priceMatrixData.length > 0 && _.get(priceMatrixData[0], 'Apttus_Config2__MatrixEntries__r.records')) {
                        const sortedDimensions = priceMatrixData.sort((n1, n2) => n1.Apttus_Config2__Sequence__c - n2.Apttus_Config2__Sequence__c);
                        let val = null;
                        for (const matrix of sortedDimensions) {
                            matrix.Apttus_Config2__MatrixEntries__r.records.forEach(entry => {
                                for (let i = 1; i <= 6; i++) {
                                    if (attribute.Apttus_Config2__Field__c === _.get(matrix, 'Apttus_Config2__Dimension' + i + 'Id__r.Apttus_Config2__Datasource__c') &&
                                        _.get(entry, 'Apttus_Config2__Dimension' + i + 'Value__c') === attributeValue) {
                                        val = PriceService.adjustValue(entry.Apttus_Config2__AdjustmentAmount__c, entry.Apttus_Config2__AdjustmentType__c, priceListItem.Apttus_Config2__ListPrice__c);
                                        break;
                                    }
                                }
                            });
                        }
                        return (val - priceListItem.Apttus_Config2__ListPrice__c);
                    } else
                        return null;
                });
            }
        });
    }

    private priceListAdjustment(price: Price, adjustmentType: string, adjustmentAmount: number, priceListItem: PriceListItem): void {
        let baseAmount = (priceListItem.Apttus_Config2__ContractPrice__c) ? priceListItem.Apttus_Config2__ContractPrice__c : priceListItem.Apttus_Config2__ListPrice__c;

        price.basePrice = baseAmount;
        price.discountPrice = PriceService.adjustValue(adjustmentAmount, adjustmentType, baseAmount);
    }

    private matrixAdjustment(p: Price, priceMatrices: Array<PriceMatrix>, attributeValue: ProductAttributeValue, quantity: number): void {
        const originalPrice = p.basePrice;

        if (priceMatrices.length > 0 && _.get(priceMatrices[0], 'Apttus_Config2__MatrixEntries__r.records')) {
            const sortedDimensions = priceMatrices.sort((n1, n2) => n1.Apttus_Config2__Sequence__c - n2.Apttus_Config2__Sequence__c);

            for (const matrix of sortedDimensions) {
                let previousValue = 0;
                for (const matrixEntry of matrix.Apttus_Config2__MatrixEntries__r.records) {
                    let validEntry: boolean = false;
                    for (let i = 1; i <= 6; i++) {
                        const prefix = 'Apttus_Config2__Dimension' + i;

                        if (_.get(matrix, prefix + 'Id__r.Apttus_Config2__ContextType__c')) {
                            if (_.get(matrix, prefix + 'Id__r.Apttus_Config2__ContextType__c') === 'Product Attribute') {
                                if (attributeValue.hasOwnProperty(_.get(matrix, prefix + 'Id__r.Apttus_Config2__Datasource__c'))) {
                                    const attrValue = _.get(attributeValue, _.get(matrix, prefix + 'Id__r.Apttus_Config2__Datasource__c'));
                                    const attrType = _.get(matrix, prefix + 'ValueType__c');
                                    validEntry = this.validateMatrixEntry(attrType, attrValue, _.get(matrixEntry, prefix + 'Value__c'), previousValue);
                                    previousValue = attrValue;
                                }
                            } else if (_.get(matrix, prefix + 'Id__r.Apttus_Config2__ContextType__c') === 'Line Item' || _.get(matrix, prefix + 'Id__r.Apttus_Config2__ContextType__c') === 'Order Line Item') {
                                const attrValue = quantity;
                                const attrType = _.get(matrix, prefix + 'ValueType__c');
                                validEntry = this.validateMatrixEntry(attrType, attrValue, _.get(matrixEntry, prefix + 'Value__c'), previousValue);
                                previousValue = attrValue;
                            }
                        }
                        if (!validEntry)
                            break;
                    }

                    if (validEntry) {
                        p.basePrice = PriceService.adjustValue(_.get(matrixEntry, 'Apttus_Config2__AdjustmentAmount__c'), _.get(matrixEntry, 'Apttus_Config2__AdjustmentType__c'), p.basePrice);
                        p.discountPrice = PriceService.adjustValue(_.get(matrixEntry, 'Apttus_Config2__AdjustmentAmount__c'), _.get(matrixEntry, 'Apttus_Config2__AdjustmentType__c'), p.discountPrice);
                    }
                }
                if (matrix.Apttus_Config2__StopProcessingMoreMatrices__c) {
                    break;
                }
            }
        }
    }

    private optionAdjustment(p: Price, productOptions: Array<ProductOptionForm>): void {
        if (productOptions) {
            productOptions.forEach(po => {
                po.productOptionComponent.Apttus_Config2__OptionPrices__r.records.forEach(price => {
                    if (!price.Apttus_Config2__PriceIncludedInBundle__c) {
                        const amount = (price.Apttus_Config2__RollupPriceMethod__c === 'Flat Price') ? price.Apttus_Config2__AdjustmentAmount__c : price.Apttus_Config2__AdjustmentAmount__c * po.quantity;
                        p.basePrice = PriceService.adjustValue(price.Apttus_Config2__AdjustmentAmount__c, price.Apttus_Config2__AdjustmentType__c, p.basePrice);
                        p.discountPrice = PriceService.adjustValue(price.Apttus_Config2__AdjustmentAmount__c, price.Apttus_Config2__AdjustmentType__c, p.discountPrice);
                    }
                });
                if (po.attributeValues) {
                    po.attributeValues.forEach(attribute => this.matrixAdjustment(p, attribute.priceMatrices, attribute.attributeValue, po.quantity));
                }
            });
        }
    }

    private validateMatrixEntry(attrType: string, attrValue: number, expectedValue: number, previousValue: number): boolean {
        if (attrType === 'Discrete')
            return attrValue === expectedValue;
        else if (attrType === 'Range')
            return attrValue >= previousValue && attrValue <= expectedValue;
        else if (attrType === 'Cumulative Range')
            return attrValue >= previousValue;
        else
            return false;
    }

    public static adjustValue(adjustmentAmount: number, adjustmentType: string, amount: number): number {
        adjustmentAmount = (isNaN(adjustmentAmount)) ? 0 : adjustmentAmount;
        let val = amount;
        if (adjustmentType === PriceService.ADJ_TYPE_PER_DISCOUNT)
            val = amount * (1 - (adjustmentAmount / 100));
        else if (adjustmentType === PriceService.ADJ_TYPE_DISCOUNT_AMOUNT)
            val = amount - adjustmentAmount;
        else if (adjustmentType === PriceService.ADJ_TYPE_PER_MARKUP)
            val = amount * (1 + (adjustmentAmount / 100));
        else if (adjustmentType === PriceService.ADJ_TYPE_MARKUP_AMT)
            val = amount + adjustmentAmount;
        else if (adjustmentType === PriceService.ADJ_TYPE_PRICE_FACTOR)
            val = amount / adjustmentAmount;
        else if (adjustmentType === PriceService.ADJ_TYPE_LIST_PRICE_OVERRIDE)
            val = adjustmentAmount;

        return val;
    }
    private static ADJ_TYPE_PER_DISCOUNT = '% Discount';
    private static ADJ_TYPE_DISCOUNT_AMOUNT = 'Discount Amount';
    private static ADJ_TYPE_PER_MARKUP = '% Markup';
    private static ADJ_TYPE_MARKUP_AMT = 'Markup Amount';
    private static ADJ_TYPE_PRICE_FACTOR = 'Price Factor';
    private static ADJ_TYPE_LIST_PRICE_OVERRIDE = 'List Price Override';

    private static FREQUENCY = {
        'Hourly': 'Per Hour',
        'Daily': 'Per Day',
        'Weekly': 'Per Week',
        'Monthly': 'Per Month',
        'Quarterly': 'Per Quarter',
        'Half Yearly': 'Per Half Year',
        'Yearly': 'Per Year'
    };
}


export class Price {
    constructor(public productPrice: number = 0
        , public basePrice: number = 0
        , public discountPrice: number = 0
        , public unitPrice: number = 0) { }
}
